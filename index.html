      function removeDuplicates (list) {
        var cache = {};
        var results = [];

        for (var i = 0; i < list.length; i++) {
          if (cache[list[i]] === undefined){
            cache[list[i]] = list[i];
            results[results.length] = list[i];
          };
        }

        return results;
      }
